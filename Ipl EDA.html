# Bowler Economy rate
bowler_stats = deliveries.groupby('bowler').agg(
    total_runs=('total_runs', 'sum'),
    balls=('ball', 'count')
).reset_index()

bowler_stats['overs'] = bowler_stats['balls'] / 6
bowler_stats['economy'] = bowler_stats['total_runs'] / bowler_stats['overs']

# Only bowlers with 500+ balls
quality = bowler_stats[bowler_stats['balls'] >= 500]
best = quality.nsmallest(10, 'economy')[['bowler', 'economy']]
print(best)
